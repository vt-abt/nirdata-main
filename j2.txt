pragma solidity ^0.8.0;

contract DataWipeCertificate {
    // Struct to represent a certificate
    struct Certificate {
        string sanitizationMethod;
        DeviceInfo deviceInfo;
        ToolInfo toolInfo;
        OperatorInfo operatorInfo;
        uint256 startTime;
        uint256 endTime;
        string verificationStatus;
        bytes32 digitalSignature; // Stores hash of signature
    }

    // Nested structs for device/tool/operator info
    struct DeviceInfo {
        string manufacturer;
        string model;
        string serial;
        string mediaType;
    }

    struct ToolInfo {
        string toolName;
        string version;
        string config;
    }

    struct OperatorInfo {
        string operatorId;
        string name;
        string location;
        string contact;
    }

    // Mapping to store certificates by unique ID
    mapping(uint256 => Certificate) public certificates;
    uint256 public certificateCount = 0;

    // Event to log new certificates
    event CertificateAdded(
        uint256 indexed certId,
        string sanitizationMethod,
        string deviceSerial,
        string operatorName
    );

    // Function to add a new certificate
    function addCertificate(
        string memory _sanitizationMethod,
        DeviceInfo memory _deviceInfo,
        ToolInfo memory _toolInfo,
        OperatorInfo memory _operatorInfo,
        uint256 _startTime,
        uint256 _endTime,
        string memory _verificationStatus,
        bytes32 _digitalSignature
    ) public {
        certificates[certificateCount] = Certificate(
            _sanitizationMethod,
            _deviceInfo,
            _toolInfo,
            _operatorInfo,
            _startTime,
            _endTime,
            _verificationStatus,
            _digitalSignature
        );
        emit CertificateAdded(
            certificateCount,
            _sanitizationMethod,
            _deviceInfo.serial,
            _operatorInfo.name
        );
        certificateCount++;
    }

    // Function to retrieve a certificate
    function getCertificate(uint256 _certId)
        public
        view
        returns (
            string memory,
            DeviceInfo memory,
            ToolInfo memory,
            OperatorInfo memory,
            uint256,
            uint256,
            string memory,
            bytes32
        )
    {
        Certificate storage cert = certificates[_certId];
        return (
            cert.sanitizationMethod,
            cert.deviceInfo,
            cert.toolInfo,
            cert.operatorInfo,
            cert.startTime,
            cert.endTime,
            cert.verificationStatus,
            cert.digitalSignature
        );
    }
}

# Test 1: Wipe an HDD and verify
1. Boot from the live USB.
2. Select the target HDD.
3. Choose "NIST SP 800-88 Purge."
4. After completion, run:
   sudo dd if=/dev/sdX | strings | grep -i "test_data"
5. If no output, pass the test.

# Test 2: Check blockchain certificate
1. Go to Sepolia Etherscan.
2. Search the contract address.
3. Verify the certificate ID matches the local PDF/JSON.

async function retrieveCertificateFromBlockchain() {
  const transactionHash = document.getElementById('transactionHash').value.trim();
  
  if (!transactionHash || transactionHash.length < 42) {
    blockchainErrorMessage.textContent = "Please enter a valid Ethereum transaction hash (minimum 42 characters)";
    blockchainErrorMessage.style.display = 'block';
    return;
  }
  
  try {
    // Use the same provider as MetaMask
    const provider = new ethers.providers.Web3Provider(window.ethereum);
    const txReceipt = await provider.getTransactionReceipt(transactionHash);
    
    if (!txReceipt) {
      throw new Error("Transaction not found or not mined yet");
    }
    
    // Get contract interface for parsing logs
    const contractInterface = new ethers.utils.Interface(contractABI);
    let certificateId = null;
    
    // Look through logs to find our CertificateAdded event
    for (const log of txReceipt.logs) {
      try {
        const parsedLog = contractInterface.parseLog(log);
        if (parsedLog && parsedLog.name === "CertificateAdded") {
          // Try multiple ways to access the certId
          if (parsedLog.args.certId !== undefined) {
            certificateId = parsedLog.args.certId;
          } else if (parsedLog.args[0] !== undefined) {
            certificateId = parsedLog.args[0];
          }
          break;
        }
      } catch (e) {
        // Not all logs will be from our contract
      }
    }
    
    if (certificateId === null) {
      throw new Error("Could not find CertificateAdded event in transaction logs");
    }
    
    // Handle BigNumber conversion properly
    let certificateIdString;
    if (typeof certificateId === 'object' && certificateId !== null && typeof certificateId.toString === 'function') {
      certificateIdString = certificateId.toString();
    } else {
      certificateIdString = String(certificateId);
    }
    
    // Show PDF viewer
    pdfViewer.style.display = 'block';
    document.getElementById('certificateId').textContent = `#${certificateIdString}`;
    
    // Update UI with loading state
    document.getElementById('pdfSanitizationMethod').textContent = "Loading...";
    document.getElementById('pdfManufacturer').textContent = "Loading...";
    document.getElementById('pdfModel').textContent = "Loading...";

pragma solidity ^0.8.0;

contract DataWipeCertificate {
    // Store encrypted certificate data
    struct EncryptedCertificate {
        bytes encryptedData; // The actual encrypted certificate
        bytes32 dataHash;    // Hash of the original plaintext data for verification
        address submittedBy; // Who submitted this certificate
        uint256 timestamp;   // When it was submitted
    }

    mapping(uint256 => EncryptedCertificate) public certificates;
    uint256 public certificateCount = 0;
    
    event CertificateAdded(uint256 certId, bytes32 dataHash);

    // Store encrypted data on blockchain
    function addEncryptedCertificate(
        bytes calldata encryptedData,
        bytes32 dataHash
    ) public {
        certificates[certificateCount] = EncryptedCertificate(
            encryptedData,
            dataHash,
            msg.sender,
            block.timestamp
        );
        
        emit CertificateAdded(certificateCount, dataHash);
        certificateCount++;
    }
    
    // Get the encrypted data (but not the key to decrypt it)
    function getEncryptedCertificate(uint256 certId)
        public
        view
        returns (bytes memory, bytes32, address, uint256)
    {
        return (
            certificates[certId].encryptedData,
            certificates[certId].dataHash,
            certificates[certId].submittedBy,
            certificates[certId].timestamp
        );
    }
}

// Add this encryption function to your frontend
async function encryptCertificateData(certificateData) {
    // Generate a random encryption key (in production, you'd manage this differently)
    const key = await window.crypto.subtle.generateKey(
        { name: "AES-GCM", length: 256 },
        true,
        ["encrypt", "decrypt"]
    );
    
    // Convert certificate data to string
    const dataString = JSON.stringify(certificateData);
    
    // Convert to ArrayBuffer
    const dataBuffer = new TextEncoder().encode(dataString);
    
    // Encrypt the data
    const iv = window.crypto.getRandomValues(new Uint8Array(12));
    const encryptedBuffer = await window.crypto.subtle.encrypt(
        { name: "AES-GCM", iv: iv },
        key,
        dataBuffer
    );
    
    // Convert to format we can store
    const encryptedData = new Uint8Array(encryptedBuffer);
    const combined = new Uint8Array(iv.length + encryptedData.length);
    combined.set(iv);
    combined.set(encryptedData, iv.length);
    
    // Calculate hash of original data for verification
    const hashBuffer = await crypto.subtle.digest('SHA-256', dataBuffer);
    const dataHash = Array.from(new Uint8Array(hashBuffer))
        .map(b => b.toString(16).padStart(2, '0'))
        .join('');
    
    // Convert to hex string for storage
    const encryptedHex = Array.from(combined)
        .map(b => b.toString(16).padStart(2, '0'))
        .join('');
    
    return {
        encryptedHex: '0x' + encryptedHex,
        dataHash: '0x' + dataHash
    };
}

async function decryptCertificateData(encryptedHex, key) {
    try {
        // Remove 0x prefix if present
        const hex = encryptedHex.startsWith('0x') ? encryptedHex.slice(2) : encryptedHex;
        
        // Convert hex to bytes
        const encryptedBytes = new Uint8Array(hex.match(/.{1,2}/g).map(byte => parseInt(byte, 16)));
        
        // Extract IV (first 12 bytes) and ciphertext
        const iv = encryptedBytes.slice(0, 12);
        const ciphertext = encryptedBytes.slice(12);
        
        // Decrypt
        const decryptedBuffer = await window.crypto.subtle.decrypt(
            { name: "AES-GCM", iv: iv },
            key,
            ciphertext
        );
        
        // Convert to string
        const decoder = new TextDecoder();
        return JSON.parse(decoder.decode(decryptedBuffer));
    } catch (error) {
        console.error("Decryption failed:", error);
        throw new Error("Failed to decrypt certificate data. Incorrect key or corrupted data.");
    }
}

// Function to retrieve and decrypt a certificate
async function retrieveAndDecryptCertificate(certId, decryptionKey) {
    try {
        // Get encrypted data from blockchain
        const [encryptedData, dataHash, , ] = await contract.getEncryptedCertificate(certId);
        
        // Decrypt the data
        const certificateData = await decryptCertificateData(encryptedData, decryptionKey);
        
        // Verify integrity by comparing hash
        const dataString = JSON.stringify(certificateData);
        const hashBuffer = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(dataString));
        const calculatedHash = '0x' + Array.from(new Uint8Array(hashBuffer))
            .map(b => b.toString(16).padStart(2, '0'))
            .join('');
        
        if (calculatedHash !== dataHash) {
            throw new Error("Certificate data has been tampered with!");
        }
        
        return certificateData;
    } catch (error) {
        console.error("Error retrieving certificate:", error);
        throw error;
    }
}
